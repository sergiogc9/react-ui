# Chip

Chips are small elements to display tags, links or useful information.

<Story
	name="playground"
	args={{
		aspectSize: 's',
		children: 'Click me',
		iconLeft: 'location',
		iconRight: 'close',
		isRightIconClickable: true,
		overlayContent: 'Edit me',
		showLeftIcon: true,
		showRightIcon: true,
		variant: 'blue'
	}}
	argTypes={{
		aspectSize: {
			control: {
				type: 'radio',
				options: ['s', 'm']
			}
		},
		variant: {
			control: {
				type: 'radio',
				options: ['transparent', 'blue', 'grey', 'yellow', 'red', 'green']
			}
		}
	}}
	parameters={getPlaygroundStoryParams({ exclude: ['onOverlayClick'] })}
>
	{({
		aspectSize,
		children,
		iconLeft,
		iconRight,
		isOverlayAlwaysVisible,
		isRightIconClickable,
		overlayContent,
		showLeftIcon,
		showRightIcon,
		variant
	}) => {
		const [isVisible, setIsVisible] = React.useState(isOverlayAlwaysVisible);
		return (
			<Chip
				aspectSize={aspectSize}
				isOverlayAlwaysVisible={isVisible}
				onClick={() => setIsVisible(!isVisible)}
				overlayContent={overlayContent}
				variant={variant}
			>
				{showLeftIcon && <Chip.Icon icon={iconLeft} styling="filled" />}
				<Chip.Label>{children}</Chip.Label>
				{showRightIcon && (
					<Chip.Icon
						icon={iconRight}
						onClick={isRightIconClickable ? () => console.log('TEST') : undefined}
						styling="outlined"
					/>
				)}
			</Chip>
		);
	}}
</Story>
