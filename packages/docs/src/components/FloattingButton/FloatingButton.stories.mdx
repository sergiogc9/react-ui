import { ArgsTable, Meta, Story, Canvas } from 'storybook/utils/components';

import FloatingButton from 'components/FloatingButton';

# FloatingButton

FloatingButton component.

<Story
	name="playground"
	args={{
		aspectSize: 'm',
		icon: 'add'
	}}
	argTypes={{
		onClick: { action: 'clicked' }
	}}
	parameters={getPlaygroundStoryParams()}
>
	{args => (
		<FloatingButton {...args}>
			<FloatingButton.Icon icon={args.icon} styling="outlined" />
		</FloatingButton>
	)}
</Story>

<Story
	name="playground with text"
	args={{
		aspectSize: 'm',
		children: 'Click me',
		icon: 'add'
	}}
	argTypes={{
		onClick: { action: 'clicked' }
	}}
	parameters={getPlaygroundStoryParams()}
>
	{args => (
		<FloatingButton {...args}>
			{args.icon && <FloatingButton.Icon icon={args.icon} styling="outlined" />}
			<FloatingButton.Text aspectSize={args.aspectSize} variant={args.variant}>
				{args.children}
			</FloatingButton.Text>
		</FloatingButton>
	)}
</Story>

## AspectSize prop

Use the `aspectSize` prop to modify the size of the Floating Button.

<Canvas>
	<Story name="s" parameters={getFixedStoryParams()}>
		<FloatingButton aspectSize="s">
			<FloatingButton.Icon icon="add" styling="outlined" />
		</FloatingButton>
	</Story>
	<Story name="m" parameters={getFixedStoryParams()}>
		<FloatingButton aspectSize="m">
			<FloatingButton.Icon icon="add" styling="outlined" />
		</FloatingButton>
	</Story>
	<Story name="l" parameters={getFixedStoryParams()}>
		<FloatingButton aspectSize="l">
			<FloatingButton.Icon icon="add" styling="outlined" />
		</FloatingButton>
	</Story>
</Canvas>

## Icon and text

The floating button can be used with or without text. Simply pass as children a `<FloatingButton.Text />` component.

<Canvas>
	<Story name="without text" parameters={getFixedStoryParams()}>
		<FloatingButton>
			<FloatingButton.Icon styling="outlined" icon="add" />
		</FloatingButton>
	</Story>
	<Story name="with text" parameters={getFixedStoryParams()}>
		<FloatingButton>
			<FloatingButton.Icon styling="outlined" icon="add" />
			<FloatingButton.Text>Click me</FloatingButton.Text>
		</FloatingButton>
	</Story>
</Canvas>
