import { ArgsTable, Meta, Story, Canvas } from 'storybook/utils/components';

# Counter

Counter component.

<Story
	args={{ numberOfItems: 100 }}
	argTypes={{
		bg: { control: { type: 'text' }, description: 'Custom background color' }
	}}
	name="playground"
	parameters={getPlaygroundStoryParams()}
>
	{args => <Counter {...args} />}
</Story>

## Sizes

The Counter size can be modified using the `aspectSize` prop:

<Canvas>
	<Story name="s simple" parameters={getFixedStoryParams()}>
		<Counter aspectSize="s" numberOfItems={2} />
	</Story>
	<Story name="s double number" parameters={getFixedStoryParams()}>
		<Counter aspectSize="s" numberOfItems={36} />
	</Story>
	<Story name="s over 1000" parameters={getFixedStoryParams()}>
		<Counter aspectSize="s" numberOfItems={1234} />
	</Story>
	<Story name="m simple" parameters={getFixedStoryParams()}>
		<Counter aspectSize="m" numberOfItems={2} />
	</Story>
	<Story name="m double number" parameters={getFixedStoryParams()}>
		<Counter aspectSize="m" numberOfItems={36} />
	</Story>
	<Story name="m over 1000" parameters={getFixedStoryParams()}>
		<Counter aspectSize="m" numberOfItems={1234} />
	</Story>
</Canvas>

Change the `Counter` styling using the different availables variants changing the `variant` prop:

	<Story name="red" parameters={getFixedStoryParams()}>

	</Story>
	<Story name="blue" parameters={getFixedStoryParams()}>
		<Counter variant="blue" numberOfItems={50} />
	</Story>

		<Counter variant="green" numberOfItems={50} />
	</Story>

		<Counter variant="yellow" numberOfItems={50} />
	</Story>
</Canvas>
