import { ArgsTable, Meta, Story, Canvas } from 'storybook/utils/components';

<Meta title="Components/CheckBox" component={CheckBox} />

# CheckBox

Radio buttons allows you to select one option from a set.

<Story name="playground" args={{}} parameters={getPlaygroundStoryParams()}>
	{args => <CheckBox {...args} />}
</Story>

## Default Selected

By default, CheckBoxs will change their status on `click` .

If you pass the `isDefaultSelected` prop to `true` it will be checked by default.

<Canvas>
	<Story name="selected" title="selected" parameters={getFixedStoryParams()}>
		<CheckBox isDefaultSelected={true} />
	</Story>
	<Story name="not selected" title="not selected" parameters={getFixedStoryParams()}>
		<CheckBox />
	</Story>
</Canvas>

## Disabled CheckBox

You can disable it adding the `isDisabled` prop.

<Canvas>
	<Story name="disabled" title="disabled" parameters={getFixedStoryParams()}>
		<CheckBox isDisabled={true} />
	</Story>
</Canvas>

## With label

<Canvas>
	<Story name="with label" title="with label" parameters={getFixedStoryParams()}>
		<CheckBox isDefaultSelected={true} label="Don't show this message again" />
	</Story>
	<Story name="with label disabled" title="with label disabled" parameters={getFixedStoryParams()}>
		<CheckBox isDisabled={true} label="Don't show this message again" />
	</Story>
</Canvas>

## With label and description

<Canvas>
	<Story name="with description" title="with description" parameters={getFixedStoryParams()}>
		<CheckBox
			isDefaultSelected={true}
			label="Don't show this message again"
			description="Remember that description prop should be always used together with label"
		/>

	</Story>
</Canvas>

## Sizes

The size only modifies the label but not the input.

<Canvas>
	<Story name="size m" title="size m" parameters={getFixedStoryParams()}>
		<CheckBox aspectSize="m" label="Don't show this message again" />
	</Story>
	<Story name="size l" title="size l" parameters={getFixedStoryParams()}>
		<CheckBox aspectSize="l" label="Don't show this message again" />
	</Story>
</Canvas>
